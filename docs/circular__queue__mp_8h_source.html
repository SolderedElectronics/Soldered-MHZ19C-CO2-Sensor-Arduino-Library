149 {
150 #ifdef ESP8266
151  esp8266::InterruptLock lock;
152 #else
153  std::lock_guard<std::mutex> lock(m_pushMtx);
154 #endif
155  const auto outPos = circular_queue<T, ForEachArg>::m_outPos.load(std::memory_order_acquire);
156  const auto inPos = circular_queue<T, ForEachArg>::m_inPos.load(std::memory_order_relaxed);
157  std::atomic_thread_fence(std::memory_order_acquire);
158  if (inPos == outPos) return circular_queue<T, ForEachArg>::defaultValue;
159  T& val = circular_queue<T, ForEachArg>::m_buffer[inPos] = std::move(circular_queue<T, ForEachArg>::m_buffer[outPos]);
160  const auto bufSize = circular_queue<T, ForEachArg>::m_bufSize;
161  std::atomic_thread_fence(std::memory_order_release);
162  circular_queue<T, ForEachArg>::m_outPos.store((outPos + 1) % bufSize, std::memory_order_relaxed);
163  circular_queue<T, ForEachArg>::m_inPos.store((inPos + 1) % bufSize, std::memory_order_release);
164  return val;

168 bool circular_queue_mp<T, ForEachArg>::for_each_requeue(const Delegate<bool(T&), ForEachArg>& fun)
169 {
170  auto inPos0 = circular_queue<T, ForEachArg>::m_inPos.load(std::memory_order_acquire);
171  auto outPos = circular_queue<T, ForEachArg>::m_outPos.load(std::memory_order_relaxed);
172  std::atomic_thread_fence(std::memory_order_acquire);
173  if (outPos == inPos0) return false;
174  do {
175  T&& val = std::move(circular_queue<T, ForEachArg>::m_buffer[outPos]);
176  if (fun(val))
177  {
178 #ifdef ESP8266
179  esp8266::InterruptLock lock;
180 #else
181  std::lock_guard<std::mutex> lock(m_pushMtx);
182 #endif
183  std::atomic_thread_fence(std::memory_order_release);
184  auto inPos = circular_queue<T, ForEachArg>::m_inPos.load(std::memory_order_relaxed);
185  std::atomic_thread_fence(std::memory_order_acquire);
186  circular_queue<T, ForEachArg>::m_buffer[inPos] = std::move(val);
187  std::atomic_thread_fence(std::memory_order_release);
188  circular_queue<T, ForEachArg>::m_inPos.store((inPos + 1) % circular_queue<T, ForEachArg>::m_bufSize, std::memory_order_release);
189  }
190  else
191  {
192  std::atomic_thread_fence(std::memory_order_release);
193  }
194  outPos = (outPos + 1) % circular_queue<T, ForEachArg>::m_bufSize;
195  circular_queue<T, ForEachArg>::m_outPos.store(outPos, std::memory_order_release);
196  } while (outPos != inPos0);
